\subsection*{Introduction}

\begin{table}[h!]
\centering
\begin{tabular}{c|c}
    \hline 
    \(X\) & \(P(X = x)\) \\
    \hline
    1 & 0.10 \\
    2 & 0.15 \\
    3 & 0.25 \\
    4 & 0.20 \\
    5 & 0.15 \\
    6 & 0.10 \\
    7 & 0.05 \\
    \hline 
\end{tabular}
\caption{Probability Mass Function of \(X\).}
\end{table}
